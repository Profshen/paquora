CREATE TABLE LIWC
(
	word VARCHAR(60) NOT NULL,
	type VARCHAR(60) NOT NULL,
	CONSTRAINT PRIMARY KEY liwc_pk(word,type)
) ENGINE=InnoDB;

CREATE TABLE LIWC_Expanded
(
	word VARCHAR(60) NOT NULL,
	type VARCHAR(60) NOT NULL,
	CONSTRAINT PRIMARY KEY liwc_pk(word,type)
) ENGINE=InnoDB;

CREATE TABLE LIWC_New_Expanded
(
	word VARCHAR(60) NOT NULL,
	type VARCHAR(60) NOT NULL,
	CONSTRAINT PRIMARY KEY liwc_pk(word,type)
) ENGINE=InnoDB;
